```javascript
// ADORAS NOTIFICATION DIAGNOSTIC SCRIPT

console.log('üîç ADORAS NOTIFICATION DIAGNOSTIC');
console.log('================================\n');

(async () => {
  const results = {
    errors: [],
    warnings: [],
    passed: [],
  };

  // Test 1: Browser Support
  console.log('1Ô∏è‚É£ Checking browser support...');
  if ('Notification' in window) {
    results.passed.push('‚úÖ Notifications API supported');
  } else {
    results.errors.push('‚ùå Notifications API NOT supported - Update browser!');
  }

  if ('serviceWorker' in navigator) {
    results.passed.push('‚úÖ Service Worker API supported');
  } else {
    results.errors.push('‚ùå Service Worker NOT supported - Update browser!');
  }

  if ('PushManager' in window) {
    results.passed.push('‚úÖ Push Manager API supported');
  } else {
    results.errors.push('‚ùå Push Manager NOT supported - Update browser!');
  }

  // Test 2: Notification Permission
  console.log('\n2Ô∏è‚É£ Checking notification permission...');
  const permission = Notification.permission;
  console.log('   Permission:', permission);
  
  if (permission === 'granted') {
    results.passed.push('‚úÖ Notification permission: GRANTED');
  } else if (permission === 'denied') {
    results.errors.push('‚ùå Notification permission: DENIED - Enable in Settings!');
  } else {
    results.warnings.push('‚ö†Ô∏è Notification permission: NOT ASKED YET');
  }

  // Test 3: Service Worker Registration
  console.log('\n3Ô∏è‚É£ Checking service worker...');
  try {
    const registration = await navigator.serviceWorker.ready;
    if (registration) {
      results.passed.push('‚úÖ Service Worker: REGISTERED');
      console.log('   SW Scope:', registration.scope);
      console.log('   SW State:', registration.active?.state);
      
      if (registration.active) {
        results.passed.push('‚úÖ Service Worker: ACTIVE');
      } else {
        results.warnings.push('‚ö†Ô∏è Service Worker: Not active yet');
      }
    }
  } catch (error) {
    results.errors.push('‚ùå Service Worker: NOT REGISTERED - Reinstall PWA!');
    console.error('   Error:', error);
  }

  // Test 4: Push Subscription
  console.log('\n4Ô∏è‚É£ Checking push subscription...');
  try {
    const registration = await navigator.serviceWorker.ready;
    const subscription = await registration.pushManager.getSubscription();
    
    if (subscription) {
      results.passed.push('‚úÖ Push Subscription: EXISTS');
      console.log('   Endpoint:', subscription.endpoint.substring(0, 50) + '...');
      console.log('   Keys:', subscription.toJSON().keys ? 'Present' : 'Missing');
    } else {
      results.errors.push('‚ùå Push Subscription: NONE - User not subscribed!');
      console.log('   üí° Solution: Menu ‚Üí Notification Settings ‚Üí Enable Notifications');
    }
  } catch (error) {
    results.errors.push('‚ùå Push Subscription check failed');
    console.error('   Error:', error);
  }

  // Test 5: VAPID Key Availability
  console.log('\n5Ô∏è‚É£ Checking VAPID keys...');
  try {
    const projectId = localStorage.getItem('supabase_project_id') || 'cyaaksjydpegofrldxbo';
    const response = await fetch(`https://${projectId}.supabase.co/functions/v1/make-server-deded1eb/notifications/vapid-public-key`, {
      headers: {
        'Authorization': `Bearer ${localStorage.getItem('adoras_access_token') || sessionStorage.getItem('adoras_access_token') || 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'}`,
      },
    });
    
    const data = await response.json();
    
    if (response.ok && data.publicKey) {
      results.passed.push('‚úÖ VAPID Public Key: AVAILABLE');
      console.log('   Key:', data.publicKey.substring(0, 20) + '...');
    } else if (data.needsSetup) {
      results.errors.push('‚ùå VAPID Keys: NOT CONFIGURED ON SERVER!');
      console.log('   üí° Solution: Add VAPID keys to Supabase Edge Functions');
      console.log('   üìñ See: /VAPID_KEYS_FOR_SUPABASE.txt');
    } else {
      results.warnings.push('‚ö†Ô∏è VAPID Key check inconclusive');
      console.log('   Response:', data);
    }
  } catch (error) {
    results.warnings.push('‚ö†Ô∏è Could not check VAPID keys (may need login)');
    console.log('   Error:', error.message);
  }

  // Test 6: Badge API (iOS)
  console.log('\n6Ô∏è‚É£ Checking Badge API (iOS)...');
  if ('setAppBadge' in navigator) {
    results.passed.push('‚úÖ Badge API: SUPPORTED');
    try {
      await navigator.setAppBadge(1);
      await navigator.clearAppBadge();
      results.passed.push('‚úÖ Badge API: WORKING');
    } catch (error) {
      results.warnings.push('‚ö†Ô∏è Badge API supported but failed - Check iOS Settings');
    }
  } else {
    results.warnings.push('‚ö†Ô∏è Badge API: NOT SUPPORTED (not iOS PWA or old iOS version)');
  }

  // Test 7: PWA Detection
  console.log('\n7Ô∏è‚É£ Checking PWA installation...');
  const isStandalone = window.matchMedia('(display-mode: standalone)').matches ||
                       window.navigator.standalone ||
                       document.referrer.includes('android-app://');
  
  if (isStandalone) {
    results.passed.push('‚úÖ PWA: INSTALLED (running standalone)');
  } else {
    results.warnings.push('‚ö†Ô∏è PWA: NOT INSTALLED (running in browser)');
    console.log('   üí° iOS: Safari ‚Üí Share ‚Üí Add to Home Screen');
    console.log('   üí° Android: Chrome will prompt to install');
  }

  // Test 8: Local Storage Check
  console.log('\n8Ô∏è‚É£ Checking local storage...');
  const hasToken = !!(localStorage.getItem('adoras_access_token') || sessionStorage.getItem('adoras_access_token'));
  const hasSeenPrompt = !!localStorage.getItem('adoras_notification_prompt_shown');
  
  if (hasToken) {
    results.passed.push('‚úÖ User: LOGGED IN');
  } else {
    results.warnings.push('‚ö†Ô∏è User: NOT LOGGED IN');
  }
  
  console.log('   Seen notification prompt:', hasSeenPrompt);
  console.log('   Access token:', hasToken ? 'Present' : 'Missing');

  // ========================================================================
  // FINAL REPORT
  // ========================================================================
  
  console.log('\n');
  console.log('================================');
  console.log('üìä DIAGNOSTIC REPORT');
  console.log('================================\n');
  
  if (results.passed.length > 0) {
    console.log('‚úÖ PASSED:');
    results.passed.forEach(msg => console.log('   ' + msg));
  }
  
  if (results.warnings.length > 0) {
    console.log('\n‚ö†Ô∏è WARNINGS:');
    results.warnings.forEach(msg => console.log('   ' + msg));
  }
  
  if (results.errors.length > 0) {
    console.log('\n‚ùå ERRORS (MUST FIX):');
    results.errors.forEach(msg => console.log('   ' + msg));
  }
  
  // ========================================================================
  // RECOMMENDATIONS
  // ========================================================================
  
  console.log('\n================================');
  console.log('üí° RECOMMENDATIONS');
  console.log('================================\n');
  
  if (results.errors.length === 0 && results.warnings.length === 0) {
    console.log('üéâ ALL SYSTEMS GO! Notifications should work!');
    console.log('\nüì± Test it:');
    console.log('   1. Menu ‚Üí Notification Settings');
    console.log('   2. Click "Send Test Notification"');
    console.log('   3. Should see notification in 2-5 seconds');
  } else {
    console.log('üîß TO FIX NOTIFICATIONS:\n');
    
    // Priority 1: VAPID Keys
    if (results.errors.some(e => e.includes('VAPID'))) {
      console.log('1Ô∏è‚É£ ADD VAPID KEYS (CRITICAL!)');
      console.log('   ‚Üí Supabase Dashboard ‚Üí Edge Functions ‚Üí Secrets');
      console.log('   ‚Üí Add keys from /VAPID_KEYS_FOR_SUPABASE.txt');
      console.log('   ‚Üí Wait 2 minutes\n');
    }
    
    // Priority 2: Permission
    if (results.errors.some(e => e.includes('DENIED'))) {
      console.log('2Ô∏è‚É£ ENABLE PERMISSIONS');
      console.log('   iOS: Settings ‚Üí Adoras ‚Üí Notifications ‚Üí ON');
      console.log('   Android: Settings ‚Üí Apps ‚Üí Adoras ‚Üí Notifications ‚Üí ON\n');
    }
    
    // Priority 3: Subscription
    if (results.errors.some(e => e.includes('Subscription'))) {
      console.log('3Ô∏è‚É£ SUBSCRIBE TO NOTIFICATIONS');
      console.log('   ‚Üí Menu ‚Üí Notification Settings');
      console.log('   ‚Üí Click "Enable Notifications"');
      console.log('   ‚Üí Allow when prompted\n');
    }
    
    // Priority 4: PWA Installation (iOS)
    if (results.warnings.some(w => w.includes('PWA')) && /iPhone|iPad|iPod/.test(navigator.userAgent)) {
      console.log('4Ô∏è‚É£ INSTALL AS PWA (iOS REQUIRED!)');
      console.log('   ‚Üí Safari ‚Üí Share ‚Üí Add to Home Screen');
      console.log('   ‚Üí Open from home screen icon (NOT Safari)');
      console.log('   ‚Üí Log in again\n');
    }
    
    // Priority 5: Service Worker
    if (results.errors.some(e => e.includes('Service Worker'))) {
      console.log('5Ô∏è‚É£ REINSTALL APP');
      console.log('   ‚Üí Delete app from home screen');
      console.log('   ‚Üí Reinstall via Add to Home Screen');
      console.log('   ‚Üí Open from home screen icon\n');
    }
  }
  
  console.log('================================');
  console.log('üìñ Full Guide: /MOBILE_NOTIFICATION_TROUBLESHOOTING.md');
  console.log('================================\n');
})();
```

## üéØ Common Results

### Result: "VAPID Keys: NOT CONFIGURED"

**Fix:**
1. Go to Supabase Dashboard
2. Edge Functions ‚Üí Secrets
3. Add 3 keys from `/VAPID_KEYS_FOR_SUPABASE.txt`
4. Wait 2 minutes
5. Run script again

---

### Result: "Permission: DENIED"

**Fix:**
- iOS: Settings ‚Üí Adoras ‚Üí Notifications ‚Üí ON
- Desktop: Browser settings ‚Üí Site permissions ‚Üí Allow

---

### Result: "Push Subscription: NONE"

**Fix:**
1. Menu ‚Üí Notification Settings
2. Click "Enable Notifications"
3. Allow when prompted
4. Run script again

---

### Result: "PWA: NOT INSTALLED"

**Fix (iOS):**
1. Safari ‚Üí Share ‚Üí Add to Home Screen
2. Open from home screen (not Safari!)
3. Log in
4. Run script again
